    var Panopto = Panopto || {};
    Panopto.GoogleAnalytics = Panopto.GoogleAnalytics || {};
    
    // Dictionary of event tracker prefixes by product id
    // These are used to keep track of the trackers we've created so far on this page instance
    // so that we can send events with the tracker prefix specified to target the specific product id
    Panopto.GoogleAnalytics.eventTrackerPrefixes = {};

    // Helper for sending a GA event to the appropriate tracker(s)
    Panopto.GoogleAnalytics.sendEvent = function (params) {
        var eventTrackerPrefix;

        if (params.productId) {
            eventTrackerPrefix = Panopto.GoogleAnalytics.eventTrackerPrefixes[params.productId];

            // Create a new tracker prefix for this product id if we haven't already
            if (!eventTrackerPrefix) {
                eventTrackerPrefix = Panopto.GoogleAnalytics.eventTrackerPrefixes[params.productId] = _.uniqueId("eventTracker");
                if (Panopto.GoogleAnalytics.sendOptInUsageData) {
                    ga("create", params.productId, 'auto', eventTrackerPrefix);
                }
            }
        }

        // Use eventTrackerPrefix if specified, otherwise use the default (page view tracker)
        var send = eventTrackerPrefix
            ? eventTrackerPrefix + ".send"
            : "send";

        // Required fields
        var fieldsObject = {
            eventCategory: params.source,
            eventAction: params.action,
            eventLabel: params.label
        };

        // Value field is optional
        if (typeof (params.value) !== "undefined") {
            fieldsObject.eventValue = params.value;
        }

        // Attach custom fields, if specified
        if (typeof (params.customFields) !== "undefined") {
            fieldsObject = _.extend(fieldsObject, params.customFields);
        }

        try {
            if (Panopto.GoogleAnalytics.sendOptInUsageData) {
                ga(send, "event", fieldsObject);
            }

            if (Panopto.GoogleAnalytics.customerTrackerPrefix) {
                ga(Panopto.GoogleAnalytics.customerTrackerPrefix + ".send", "event", fieldsObject);
            }
        }
        catch (e) {
            // Error sending analytics data, so do nothing
        }
    };
